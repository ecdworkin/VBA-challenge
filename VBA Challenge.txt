Sub ChallengeTicker()    'Declare necessary variables        Dim Ticker As String 'tickers?high/lows        Dim StartRow As Double        Dim EndRow As Double        Dim YearlyChange As Double        Dim PercentChange As Double        Dim Volume As Double        Dim i As Double        Dim j As Double        Dim ws As Worksheet        Dim lastrow As Double        Dim MaxDecrease As Double        Dim MaxIncrease As Double        Dim MaxVolume As Double                For Each ws In Worksheetsws.Activatelastrow = Cells(Rows.Count, "a").End(xlUp).Row'Establish headersCells(1, "i").Value = "Ticker"Cells(1, "j").Value = "Yearly Change"Cells(1, "k").Value = "Percent Change"Cells(1, "l").Value = "Stock Volume"        'Create a loop that goes through all the stocks for one year and outputs the following        'ticker symbol            For i = 2 To lastrow                If Cells(i, 1).Value <> Cells(i - 1, 1).Value Then                OpenPrice = Format(Cells(i, 3).Value, "#.00")                StartRow = Cells(i, 1).Row                Ticker = Cells(i, 1).Value                Cells(Rows.Count, "i").End(xlUp).Offset(1, 0).Activate                ActiveCell.Value = Ticker                End If                If Cells(i, 1).Value <> Cells(i + 1, 1).Value Then                ClosePrice = Format(Cells(i, 6).Value, "#.00")                EndRow = Cells(i, 1).Row                Volume = Application.WorksheetFunction.Sum(Range(Cells(StartRow, "g"), Cells(EndRow, "g")))                        'yearly change in price - opening at beginning of year closing at end of year                YearlyChange = ClosePrice - OpenPrice                        '% Change in price from opening at beginning of year to closing at end of year                PercentChange = (ClosePrice - OpenPrice) / OpenPrice                'Landing above calcs                Cells(Rows.Count, "j").End(xlUp).Offset(1, 0).Activate                ActiveCell.Value = Format(YearlyChange, "#.00")                Cells(Rows.Count, "k").End(xlUp).Offset(1, 0).Activate                ActiveCell.Value = Format(PercentChange, "0.00%")                Cells(Rows.Count, "l").End(xlUp).Offset(1, 0).Activate                ActiveCell.Value = Volume                End If                Next i                                Range("i:l").EntireColumn.AutoFit                        lastrow2 = Cells(Rows.Count, "i").End(xlUp).Row        For j = 2 To lastrow2                If Cells(j, "j").Value > 0 Then        Cells(j, "j").Interior.Color = vbGreen        Else        Cells(j, "j").Interior.Color = vbRed        End If        Next j                Range("n2").Value = "Greatest Percent Increase"        Range("n3").Value = "Greatest Percent Decrease"        Range("n4").Value = "Greatest Total Volume"        Range("o1").Value = "Ticker"        Range("p1").Value = "Value"        Range("p2").Value = Format(Application.WorksheetFunction.Max(Range(Cells(2, "k"), Cells(lastrow2, "k"))), "0.00%")        Range("p3").Value = Format(Application.WorksheetFunction.Min(Range(Cells(2, "k"), Cells(lastrow2, "k"))), "0.00%")        Range("p4").Value = Application.WorksheetFunction.Max(Range(Cells(2, "l"), Cells(lastrow2, "l")))                MaxIncrease = Range("p2").Value        MaxIncrease = MaxIncrease * 100        Range("k:k").Find(MaxIncrease).Activate        Ticker = ActiveCell.Offset(0, -2).Value        Range("o2").Value = Ticker                MaxDecrease = Range("p3").Value        MaxDecrease = MaxDecrease * 100        Range("k:k").Find(MaxDecrease).Activate        Ticker = ActiveCell.Offset(0, -2).Value        Range("o3").Value = Ticker                MaxVolume = Range("p4").Value        Range("l:l").Find(MaxVolume).Activate        Ticker = ActiveCell.Offset(0, -3).Value        Range("o4").Value = Ticker        Range("n:p").EntireColumn.AutoFit                Next ws            'Add README to RepositoryEnd Sub
